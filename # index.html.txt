function fillSelect(el, data){ el.innerHTML=data.map(([v,t])=>`<option value="${v}">${t}</option>`).join(''); }
function fillSimple(el, data){ el.innerHTML=data.map(v=>`<option value="${v}">${v}</option>`).join(''); }
function findTerm(text){ const s=text.trim().toLowerCase(); return JARGONS.find(j=>s.includes(j.term.toLowerCase())); }
function translateFallback(text,to){ return `${text} (${to.toUpperCase()} preview)`; }

function renderStats(){ els.jargonCount.textContent = String(JARGONS.length); els.langCount.textContent = String(LANGUAGES.length); }
function renderChips(){
  els.chips.innerHTML = INDUSTRIES.map(i=>`<button class="chip ${i===els.ind.value?'active':''}" data-i="${i}">${i}</button>`).join('');
  [...els.chips.querySelectorAll('button')].forEach(b=>b.onclick=()=>{els.ind.value=b.dataset.i; renderChips(); renderGlossary();});
}
function renderGlossary(){
  const list = JARGONS.filter(j=>j.industry===els.ind.value);
  els.glossary.innerHTML = list.map(j=>`<div class="term"><b>${j.term}</b>${j.explanation}</div>`).join('');
}
function applyUiLanguage(){
  const lang = els.uiLang.value;
  const copy = UI_COPY[lang] || UI_COPY.en;
  els.heroTitle.textContent = copy.title;
  els.heroSub.textContent = copy.sub;
}

function translateAndExplain(){
  els.error.textContent = '';
  const text = els.input.value.trim();
  if(!text){ els.error.textContent = 'Please enter a finance term or short sentence.'; return; }
  const known = findTerm(text);
  const to = els.to.value;
  const industry = els.ind.value;
  const translated = known ? (known.tr[to] || known.tr.en) : translateFallback(text,to);
  const explain = known ? known.explanation : `In ${industry}, this appears to be a performance/risk/profitability term used in reporting and decision making.`;
  const confidence = known ? 'High (glossary matched)' : 'Medium (fallback)';
  els.out.innerHTML = `<strong>Industry:</strong> ${industry}<br><strong>Translation:</strong> ${translated}<br><strong>Explanation:</strong> ${explain}<br><strong>Confidence:</strong> ${confidence}`;
}

function suggestTerms(){
  const industry = els.ind.value;
  const terms = relatedByIndustry[industry] || [];
  els.out.innerHTML = `<strong>Related ${industry} terms:</strong><br>${terms.map((t,i)=>`${i+1}. ${t}`).join('<br>')}`;
  els.error.textContent='';
}

(function init(){
  fillSelect(els.uiLang, LANGUAGES);
  fillSelect(els.from, LANGUAGES);
  fillSelect(els.to, LANGUAGES);
  fillSimple(els.ind, INDUSTRIES);
  els.uiLang.value='en'; els.from.value='en'; els.to.value='es'; els.ind.value='Banking';
  renderStats(); renderChips(); renderGlossary(); applyUiLanguage();

  document.getElementById('translateBtn').onclick = translateAndExplain;
  document.getElementById('suggestBtn').onclick = suggestTerms;
  document.getElementById('swapBtn').onclick = ()=>{ const t=els.from.value; els.from.value=els.to.value; els.to.value=t; };
  els.ind.onchange = ()=>{ renderChips(); renderGlossary(); };
  els.uiLang.onchange = applyUiLanguage;
})();
